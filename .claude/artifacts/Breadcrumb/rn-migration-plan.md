## Implementation: Breadcrumb.native.tsx

Replace the stub entirely. Key changes from the web version:

1. Remove `as="nav"`, `as="ol"`, `as="li"`, `as="span"` props — RN BaseBox ignores them and they trigger warnings
2. Add `flexDirection="row"` to all horizontal containers (RN defaults to column)
3. Remove `style={listStyleNone}` — irrelevant in RN
4. On the outer `BaseBox`, pass `{...makeAccessible({ role: 'navigation', label: accessibilityLabel })}` to preserve semantic navigation landmark (use `accessibilityRole` instead of HTML `nav`)
5. On the list `BaseBox`, remove the `makeAccessible({ label })` call that was on the `ol` — the label moves to the outer nav wrapper
6. On each item `BaseBox`, remove `makeAccessible({ current: ariaCurrent })` — `aria-current` has no direct RN equivalent; acceptable to omit or use `accessibilityState` if needed
7. The Separator component: remove `as="span"` from `Text`, no other changes needed
8. `React.forwardRef` pattern: keep — `BladeElementRef` ref forwarding works the same

Snippet outline:
```tsx
import React from 'react';
import type { BreadcrumbProps } from './types';
import { BreadcrumbContext } from './BreadcrumbContext';
import BaseBox from '~components/Box/BaseBox';
import { Text } from '~components/Typography';
import { makeAccessible } from '~utils/makeAccessible';
import { getStyledProps } from '~components/Box/styledProps';
import { metaAttribute, MetaConstants } from '~utils/metaAttribute';
import type { BladeElementRef } from '~utils/types';
import { makeAnalyticsAttribute } from '~utils/makeAnalyticsAttribute';

const Separator = ({ size, color }: Pick<BreadcrumbProps, 'size' | 'color'>) => (
  <Text
    color={color === 'white' ? 'surface.text.staticWhite.muted' : 'surface.text.gray.muted'}
    size={size}
    weight="medium"
  >
    /
  </Text>
);

const _Breadcrumb = ({ size = 'medium', color = 'primary', showLastSeparator = false,
  accessibilityLabel = 'Breadcrumb', children, ...rest }: BreadcrumbProps,
  ref: React.Ref<BladeElementRef>
) => {
  const contextValue = React.useMemo(() => ({ size, color }), [size, color]);
  return (
    <BaseBox
      ref={ref as never}
      {...getStyledProps(rest)}
      {...metaAttribute({ name: MetaConstants.Breadcrumb })}
      {...makeAnalyticsAttribute(rest)}
      {...makeAccessible({ role: 'navigation' })}
    >
      <BreadcrumbContext.Provider value={contextValue}>
        <BaseBox
          display="flex"
          flexDirection="row"    // REQUIRED: RN default is column
          flexWrap="wrap"
          gap="spacing.2"
          alignItems="center"
          {...makeAccessible({ label: accessibilityLabel })}
        >
          {React.Children.map(children, (child, index) => (
            <BaseBox
              key={index}
              display="flex"
              flexDirection="row"  // REQUIRED: RN default is column
              alignItems="center"
              gap="spacing.2"
              {...metaAttribute({ name: MetaConstants.BreadcrumbItem })}
            >
              {child}
              <BaseBox>
                {index !== React.Children.count(children) - 1 && <Separator size={size} />}
                {index === React.Children.count(children) - 1 && showLastSeparator && (
                  <Separator size={size} color={color} />
                )}
              </BaseBox>
            </BaseBox>
          ))}
        </BaseBox>
      </BreadcrumbContext.Provider>
    </BaseBox>
  );
};

const Breadcrumb = React.forwardRef(_Breadcrumb);
export { Breadcrumb };
```

## Implementation: BreadcrumbItem.native.tsx

Nearly identical to web version. Key differences:

1. `BaseBox` container: add `flexDirection="row"` (used in the `isCurrentPage` branch)
2. `BaseLink` works on native — no changes needed to the non-`isCurrentPage` branch
3. `opacity` token import from `~tokens/global` — no change needed
4. Remove no web-only props — `BreadcrumbItemProps` has none

Snippet outline:
```tsx
import React from 'react';
import type { BreadcrumbItemProps } from './types';
import { BreadcrumbContext } from './BreadcrumbContext';
import { BaseLink } from '~components/Link/BaseLink';
import { Text } from '~components/Typography';
import BaseBox from '~components/Box/BaseBox';
import { opacity } from '~tokens/global';
import { makeAnalyticsAttribute } from '~utils/makeAnalyticsAttribute';

const BreadcrumbItem = ({ children, href, icon: Icon, isCurrentPage,
  onClick, accessibilityLabel, ...rest }: BreadcrumbItemProps) => {
  const { color, size } = React.useContext(BreadcrumbContext);

  if (isCurrentPage) {
    const isWhite = color === 'white';
    return (
      <BaseBox display="flex" flexDirection="row" gap="spacing.2" alignItems="center">
        {Icon && (
          <Icon size={size}
            color={isWhite ? 'surface.icon.staticWhite.normal' : 'surface.icon.gray.normal'}
          />
        )}
        <Text weight="medium" size={size}
          color={isWhite ? 'surface.text.staticWhite.normal' : 'surface.text.gray.normal'}
        >
          {children}
        </Text>
      </BaseBox>
    );
  }

  return (
    <BaseLink
      size={size}
      color={color}
      opacity={color !== 'primary' ? opacity[700] : 1}
      icon={Icon}
      href={href}
      onClick={onClick}
      accessibilityProps={{ label: accessibilityLabel }}
      {...makeAnalyticsAttribute(rest)}
    >
      {children ?? ''}
    </BaseLink>
  );
};

export { BreadcrumbItem };
```

## Implementation: Breadcrumb.native.test.tsx

```tsx
import React from 'react';
import { fireEvent } from '@testing-library/react-native';
import { Breadcrumb, BreadcrumbItem } from '../';
import renderWithTheme from '~utils/testing/renderWithTheme.native';

beforeAll(() => jest.spyOn(console, 'error').mockImplementation());
afterAll(() => jest.restoreAllMocks());

describe('<Breadcrumb />', () => {
  it('should render', () => {
    const { toJSON } = renderWithTheme(
      <Breadcrumb>
        <BreadcrumbItem href="/">Home</BreadcrumbItem>
        <BreadcrumbItem href="/about">About</BreadcrumbItem>
        <BreadcrumbItem href="/contact">Contact</BreadcrumbItem>
      </Breadcrumb>
    );
    expect(toJSON()).toMatchSnapshot();
  });

  it('should render current page item as text', () => {
    const { toJSON, getByText } = renderWithTheme(
      <Breadcrumb>
        <BreadcrumbItem href="/">Home</BreadcrumbItem>
        <BreadcrumbItem href="/contact" isCurrentPage>Contact</BreadcrumbItem>
      </Breadcrumb>
    );
    expect(toJSON()).toMatchSnapshot();
    // current page renders as Text, not a Pressable link
    expect(getByText('Contact')).toBeTruthy();
  });

  it('should work with showLastSeparator', () => {
    const { toJSON } = renderWithTheme(
      <Breadcrumb showLastSeparator>
        <BreadcrumbItem href="/">Home</BreadcrumbItem>
        <BreadcrumbItem href="/contact">Contact</BreadcrumbItem>
      </Breadcrumb>
    );
    expect(toJSON()).toMatchSnapshot();
  });

  it('should call onClick when BreadcrumbItem is pressed', () => {
    const onClick = jest.fn();
    const { getByText } = renderWithTheme(
      <Breadcrumb>
        <BreadcrumbItem onClick={onClick} href="/home">Home</BreadcrumbItem>
      </Breadcrumb>
    );
    fireEvent.press(getByText('Home'));
    expect(onClick).toHaveBeenCalledTimes(1);
  });

  it('should render with icon', () => {
    // Use any blade icon, e.g. HomeIcon — replace with available icon in test
    const { toJSON } = renderWithTheme(
      <Breadcrumb>
        <BreadcrumbItem href="/" isCurrentPage>Home</BreadcrumbItem>
      </Breadcrumb>
    );
    expect(toJSON()).toMatchSnapshot();
  });
});
```

## Style Translation

No `Styled*.web.tsx` files exist. All layout uses `BaseBox` props which map directly to RN style attributes. The only translation needed is:

| Web Pattern | Native Change |
|-------------|--------------|
| `as="nav"` on BaseBox | Remove; add `makeAccessible({ role: 'navigation' })` |
| `as="ol"` on BaseBox | Remove |
| `as="li"` on BaseBox | Remove |
| `as="span"` on BaseBox / Text | Remove |
| `style={listStyleNone}` | Remove |
| All flex containers missing `flexDirection` | Add `flexDirection="row"` |

## Decisions

| # | Decision | Choice | Rationale |
|---|----------|--------|-----------|
| 1 | File strategy | Replace stubs in-place | Stubs already exist; no rename needed |
| 2 | `aria-current` on native | Omit | No direct RN equivalent; accessibility role covers navigation semantics |
| 3 | `accessibilityLabel` placement | Move to inner list BaseBox | Mirrors how `ol[aria-label]` semantics work; outer gets `role="navigation"` |
| 4 | Separator aria-hidden | Omit | Screen readers handle decorative text differently on RN; the `/` separator has no semantic meaning |
| 5 | `flexDirection` | Explicitly set `"row"` on ALL horizontal flex containers | RN defaults to column; missing this causes layout breakage |
