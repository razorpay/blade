export const Spacing = () => {
  const { theme } = useTheme();

    <table>
      <tbody>

          <tr key={key}>
            <td>{`theme.spacing.${key}`}</td>
            <td>
              <div
                style={{
                  height: `${value}px`,

                }}
              />
            </td>
          </tr>
        ))}
      </tbody>
    </table>
  );
};
