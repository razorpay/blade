import { Meta, DocsContainer } from '@storybook/addon-docs';
import { useTheme, BladeProvider } from '@razorpay/blade/components';
import { paymentTheme, bankingTheme } from '@razorpay/blade/tokens';

<Meta
  title="Tokens/Spacing"
  parameters={{
    docs: {
      container: ({ children, context }) => {
        const getThemeTokens = () => {
          if (context.globals.themeTokens === 'paymentTheme') {
            return paymentTheme;
          }
          if (context.globals.themeTokens === 'bankingTheme') {
            return bankingTheme;
          }
        };
        return (
          <DocsContainer context={context}>
            <BladeProvider
              key={`${context.globals.themeTokens}-${context.globals.colorScheme}`}
              themeTokens={getThemeTokens()}
              colorScheme={context.globals.colorScheme}
            >
              {children}
            </BladeProvider>
          </DocsContainer>
        )
      }
    },
  }}
/>

# ↔️ Spacing

export const Spacing = () => {
  const { theme } = useTheme();
  return (
    <>
      <table>
        <tbody>
          {Object.entries(theme.spacing).map(([key, value]) => (
            <tr key={key}>
              <td>{`theme.spacing.${key}`}</td>
              <td><div style={{height: `${value}px`, width: `${value}px`, backgroundColor:'rgba(223, 135, 0, 0.18)'}}></div></td>
            </tr>
          ))}
        </tbody>
      </table>
    </>
  );
};

<Spacing />
